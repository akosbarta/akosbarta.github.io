whichone <- function(interval) {
  from <- interval[1]
  to <- interval[2]
  valuegrowth <- c()
  functionnames <- c("linear black","sqrt blue","power ^2 red","power ^3 green","exponential orange")
  valuegrowth <- c(valuegrowth, to-from)
  valuegrowth <- c(valuegrowth, sqrt(to)-sqrt(from))
  valuegrowth <- c(valuegrowth, to^2-from^2)
  valuegrowth <- c(valuegrowth, to^3-from^3)
  valuegrowth <- c(valuegrowth, exp(to)-exp(from))
  functionnames[which(valuegrowth == max(valuegrowth))]
}
whichone(c(.8,1.2))
for (i in seq(.01,2,by=.01)) {
  printit <- paste(whichone(c(i-0.01,i)), i)
  print(printit)
}

relDiff <- function(x1=100,x2,x3) {
  if (is.numeric(x1)&&is.numeric(x2)&&is.numeric(x3)) {
    c((x2-x1)/x1,(x3-x2)/x2)
  }
}

relDiff <- function(sequence) {
  returned <- c()
  for(i in seq(2,length(sequence))) {
    returned <- paste(returned,round((sequence[i] - sequence[i-1])/sequence[i-1],3))
  }
  returned
}

relDiff(seq(1,10,by=1))

  if (x > 0 && is.numeric(x) && is.numeric(a) && is.finite(r)) {

#Problem 4
polyEval <- function(x,a) {
  if (is.numeric(x) && is.vector(a)) {
    result <- 0
    for (ai in seq(0,length(a)-1)) {
      result <- a[ai+1]*x^(ai) + result
    }
    result
  }
}

polyEval(2, c(2,3,1))

}

  ifelse(i <= 20,i*.15,ifelse(i > 30,20*.15 + 10*.3 + (i-30)*.6,20*.15 + (i-20)*.3))
}
